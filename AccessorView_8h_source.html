202  void create(const Model& model, const Accessor& accessor) {
203  const CesiumGltf::BufferView* pBufferView =
204  Model::getSafe(&model.bufferViews, accessor.bufferView);
205  if (!pBufferView) {
206  this->_status = AccessorViewStatus::InvalidBufferViewIndex;
207  return;
208  }
209 
210  const CesiumGltf::Buffer* pBuffer =
211  Model::getSafe(&model.buffers, pBufferView->buffer);
212  if (!pBuffer) {
213  this->_status = AccessorViewStatus::InvalidBufferIndex;
214  return;
215  }
216 
217  const std::vector<std::byte>& data = pBuffer->cesium.data;
218  int64_t bufferBytes = int64_t(data.size());
219  if (pBufferView->byteOffset + pBufferView->byteLength > bufferBytes) {
220  this->_status = AccessorViewStatus::BufferTooSmall;
221  return;
222  }
223 
224  int64_t accessorByteStride = accessor.computeByteStride(model);
225  int64_t accessorComponentElements = accessor.computeNumberOfComponents();
226  int64_t accessorComponentBytes = accessor.computeByteSizeOfComponent();
227  int64_t accessorBytesPerStride =
228  accessorComponentElements * accessorComponentBytes;
229 
230  if (sizeof(T) != accessorBytesPerStride) {
231  this->_status = AccessorViewStatus::WrongSizeT;
232  return;
233  }
234 
235  int64_t accessorBytes = accessorByteStride * accessor.count;
236  int64_t bytesRemainingInBufferView =
237  pBufferView->byteLength -
238  (accessor.byteOffset + accessorByteStride * (accessor.count - 1) +
239  accessorBytesPerStride);
240  if (accessorBytes > pBufferView->byteLength ||
241  bytesRemainingInBufferView < 0) {
242  this->_status = AccessorViewStatus::BufferViewTooSmall;
243  return;
244  }
245 
246  this->_pData = pBuffer->cesium.data.data();
247  this->_stride = accessorByteStride;
248  this->_offset = accessor.byteOffset + pBufferView->byteOffset;
249  this->_size = accessor.count;
250  this->_status = AccessorViewStatus::Valid;
251  }
