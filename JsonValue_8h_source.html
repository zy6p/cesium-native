30 constexpr T losslessNarrowOrDefault(U u, T defaultValue) noexcept {
31  constexpr const bool is_different_signedness =
32  (std::is_signed<T>::value != std::is_signed<U>::value);

34  const T t = gsl::narrow_cast<T>(u);
35 
36  if (static_cast<U>(t) != u ||
